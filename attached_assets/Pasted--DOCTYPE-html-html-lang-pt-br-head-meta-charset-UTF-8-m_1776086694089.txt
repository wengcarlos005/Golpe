<script>
    // --- FIREBASE CONFIG ---
    const firebaseConfig = {
        apiKey: "AIzaSyCj6Q8eWBiiBXOx_7vOICzGHjBSDYFNV2Q",
        databaseURL: "https://golpe-63ee3-default-rtdb.firebaseio.com",
        projectId: "golpe-63ee3",
    };
    firebase.initializeApp(firebaseConfig);
    const db = firebase.database();

    let minhaInfo = { nome: "", id: "", sala: "", isHost: false };
    let alvoId = null;

    function nav(id) {
        document.querySelectorAll('.screen').forEach(s => s.classList.remove('active'));
        document.getElementById(id).classList.add('active');
    }

    // --- SALAS ---
    function criarSala() {
        const n = document.getElementById('input-nome').value;
        if(!n) return alert("Digite um nome!");
        const cod = Math.random().toString(36).substring(2, 6).toUpperCase();
        minhaInfo = { nome: n, id: "p_"+Date.now(), sala: cod, isHost: true };
        db.ref(`salas/${cod}`).set({ iniciado: false, host: minhaInfo.id, fase: "ESPERA", log: [] });
        conectar();
    }

    function entrarSala() {
        const n = document.getElementById('input-nome').value;
        const cod = document.getElementById('input-sala-cod').value.toUpperCase();
        if(!n || !cod) return alert("Preencha tudo!");
        minhaInfo = { nome: n, id: "p_"+Date.now(), sala: cod, isHost: false };
        conectar();
    }

    function conectar() {
        db.ref(`salas/${minhaInfo.sala}/jogadores/${minhaInfo.id}`).set({
            nome: minhaInfo.nome, moedas: 2, vivo: true, host: minhaInfo.isHost
        });
        document.getElementById('display-cod').innerText = minhaInfo.sala;
        nav('screen-lobby');
        ouvirSala();
    }

    function ouvirSala() {
        db.ref(`salas/${minhaInfo.sala}`).on('value', snap => {
            const d = snap.val();
            if(!d) return;
            if(d.iniciado && document.getElementById('screen-lobby').classList.contains('active')) nav('screen-mesa');
            renderLobby(d);
            if(d.iniciado) renderMesa(d);
        });
    }

    function renderLobby(d) {
        const jogs = d.jogadores || {};
        const lista = document.getElementById('lista-jogadores');
        lista.innerHTML = "";
        Object.values(jogs).forEach(j => {
            lista.innerHTML += `<div style="padding:5px 0;">• ${j.nome} ${j.host ? '<span style="color:var(--gold); font-size:0.8em;">[HOST]</span>' : ''}</div>`;
        });
        const btn = document.getElementById('btn-iniciar');
        if(minhaInfo.isHost) {
            btn.disabled = Object.keys(jogs).length < 2;
            btn.innerText = btn.disabled ? "AGUARDANDO PLAYERS..." : "INICIAR JOGO";
        } else {
            btn.innerText = "AGUARDANDO HOST...";
        }
    }

    // --- TEMPO E RESOLUÇÃO AUTOMÁTICA ---
    setInterval(() => {
        if (!minhaInfo.sala) return;
        db.ref(`salas/${minhaInfo.sala}`).once("value", snap => {
            const d = snap.val();
            if (!d || !d.iniciado || !d.timerFinal) return;

            const agora = Date.now();
            const restante = d.timerFinal - agora;
            const total = (d.fase === "ACAO" ? 30000 : 10000);
            
            document.getElementById("timer-fill").style.width = Math.max(0, (restante / total) * 100) + "%";
            document.getElementById("label-tempo").innerText = `00:${Math.max(0, Math.floor(restante/1000)).toString().padStart(2, '0')}`;

            if (restante <= 0) {
                if (d.fase !== "ACAO") {
                    if(d.vezDe === minhaInfo.id) executarEfeito(); // Autopass
                } else if (d.vezDe === minhaInfo.id) {
                    passarTurno(); // Autolose turn
                }
            }
        });
    }, 500);

    // --- JOGO ---
    function iniciarPartida() {
        let b = []; ["Político","Bicheiro","X9","Juiz","Miliciano","Segurança"].forEach(p => { for(let i=0; i<3; i++) b.push(p); });
        b.sort(() => Math.random() - 0.5);
        db.ref(`salas/${minhaInfo.sala}/jogadores`).once("value", s => {
            const js = s.val();
            const ids = Object.keys(js);
            let up = { iniciado: true, vezDe: ids[0], fase: "ACAO", timerFinal: Date.now() + 30000, baralho: b };
            ids.forEach(id => {
                up[`jogadores/${id}/cartas`] = [b.pop(), b.pop()];
                up[`jogadores/${id}/moedas`] = 2;
            });
            db.ref(`salas/${minhaInfo.sala}`).update(up);
        });
    }

    function renderMesa(d) {
        const jogs = d.jogadores || {};
        
        // Verificação de Vitória
        const ativos = Object.keys(jogs).filter(id => (jogs[id].cartas || []).length > 0);
        if(ativos.length === 1) {
            document.getElementById('area-botoes').innerHTML = `<div style="background:var(--gold); color:black; padding:20px; border-radius:10px; text-align:center; font-weight:bold;">🏆 ${jogs[ativos[0]].nome.toUpperCase()} VENCEU!</div>`;
            return;
        }

        // Render Adversários
        let advH = "";
        for(let id in jogs) {
            if(id === minhaInfo.id) continue;
            const vivo = (jogs[id].cartas || []).length > 0;
            let st = (id === d.vezDe ? "active-turn" : (id === alvoId ? "target-selected" : ""));
            advH += `<div class="player-card ${st}" style="opacity:${vivo?1:0.3}" onclick="${vivo ? `alvoId='${id}'` : ''}">
                <small>${jogs[id].nome}</small><br>${vivo ? '💰'+jogs[id].moedas : 'ELIMINADO'}
            </div>`;
        }
        document.getElementById('adversarios').innerHTML = advH;

        // Log
        const logCont = document.getElementById('chat-log');
        const hist = d.log || [];
        logCont.innerHTML = `<div style="font-weight:bold; color:var(--gold); margin-bottom:10px;">HISTÓRICO</div>`;
        hist.slice(-8).forEach(msg => {
            logCont.innerHTML += `<div class="action-msg"><span>${msg}</span></div>`;
        });

        // Meu Status
        const eu = jogs[minhaInfo.id];
        document.getElementById('minhas-moedas').innerText = `💰 ${eu.moedas}`;
        document.getElementById('minhas-cartas').innerHTML = (eu.cartas || []).map(c => `<div class="minha-carta">${c}</div>`).join("");
        document.getElementById('status-fase').innerText = d.fase;

        // Botões
        const area = document.getElementById('area-botoes');
        area.innerHTML = "";
        if((eu.cartas || []).length === 0) return area.innerHTML = "<p>Você está fora do jogo.</p>";

        if(d.fase === "ACAO" && d.vezDe === minhaInfo.id) {
            area.innerHTML += `<button class="btn-yellow" onclick="pedirAcao('Político')">Político (+3 💰)</button>`;
            area.innerHTML += `<button class="btn-yellow" onclick="pedirAcao('Propina')">Propina (Rouba 3)</button>`;
            area.innerHTML += `<button class="btn-yellow" onclick="pedirAcao('X9')">Investigar (X9)</button>`;
            area.innerHTML += `<button class="btn-yellow" onclick="pedirAcao('Trabalhar')" style="background:#444; color:white;">Trabalhar (+1 💰)</button>`;
            if(eu.moedas >= 7) area.innerHTML += `<button class="btn-yellow" onclick="pedirAcao('Golpe')" style="background:#b71c1c; color:white;">DAR GOLPE (-7 💰)</button>`;
        } else if(d.fase === "DUVIDA" && d.acaoPendente.autorId !== minhaInfo.id) {
            area.innerHTML += `<button class="btn-yellow" onclick="reagir('DUVIDO')">DUVIDAR!</button>`;
            area.innerHTML += `<button class="btn-yellow" onclick="reagir('PASSAR')" style="background:#444; color:white;">ACEITAR</button>`;
        } else if(d.fase === "BLOQUEIO" && d.acaoPendente.alvoId === minhaInfo.id) {
            area.innerHTML += `<button class="btn-yellow" onclick="reagir('BLOQUEIO')">BLOQUEAR</button>`;
            area.innerHTML += `<button class="btn-yellow" onclick="reagir('PASSAR')" style="background:#444; color:white;">ACEITAR</button>`;
        } else if(d.fase === "REVELAR_ESCOLHA" && d.acaoPendente.alvoId === minhaInfo.id) {
            eu.cartas.forEach((c, i) => {
                area.innerHTML += `<button class="btn-yellow" onclick="mostrarCartaAoX9(${i})">Mostrar ${c}</button>`;
            });
        }
    }

    function pedirAcao(tipo) {
        if(["Propina","X9","Golpe"].includes(tipo) && !alvoId) return alert("Selecione um alvo!");
        if(tipo === "Propina") {
             db.ref(`salas/${minhaInfo.sala}/jogadores/${minhaInfo.id}/moedas`).once('value', s => {
                 if(s.val() < 4) return alert("Propina exige 4 moedas de entrada!");
                 processarAcao(tipo);
             });
        } else processarAcao(tipo);
    }

    function processarAcao(tipo) {
        let f = (tipo === "Trabalhar" || tipo === "Golpe") ? "EXECUCAO" : "DUVIDA";
        db.ref(`salas/${minhaInfo.sala}`).update({
            acaoPendente: { autorId: minhaInfo.id, tipo, alvoId: alvoId || null },
            fase: f, timerFinal: Date.now() + (f === "DUVIDA" ? 10000 : 30000)
        });
        db.ref(`salas/${minhaInfo.sala}/log`).once('value', s => {
            let l = s.val() || []; l.push(`@${minhaInfo.nome} usou ${tipo}`);
            db.ref(`salas/${minhaInfo.sala}/log`).set(l);
        });
        if(f === "EXECUCAO") executarEfeito();
    }

    function reagir(r) {
        db.ref(`salas/${minhaInfo.sala}`).once('value', s => {
            const d = s.val(); const a = d.acaoPendente;
            if(r === "DUVIDO") {
                let tem = d.jogadores[a.autorId].cartas.includes(a.tipo);
                tem ? perderCarta(minhaInfo.id, executarEfeito) : perderCarta(a.autorId, passarTurno);
            } else if(r === "BLOQUEIO") {
                db.ref(`salas/${minhaInfo.sala}`).update({ fase: "DUVIDA_BLOQUEIO", timerFinal: Date.now() + 10000 });
            } else {
                if(d.fase === "DUVIDA" && ["Propina","X9"].includes(a.tipo)) {
                    db.ref(`salas/${minhaInfo.sala}`).update({ fase: "BLOQUEIO", timerFinal: Date.now() + 10000 });
                } else {
                    if(a.tipo === "X9") db.ref(`salas/${minhaInfo.sala}`).update({ fase: "REVELAR_ESCOLHA", timerFinal: Date.now() + 10000 });
                    else executarEfeito();
                }
            }
        });
    }

    function executarEfeito() {
        db.ref(`salas/${minhaInfo.sala}`).once('value', s => {
            const d = s.val(); const a = d.acaoPendente; if(!a) return;
            let up = {}; const aut = d.jogadores[a.autorId];
            if(a.tipo === "Político") up[`jogadores/${a.autorId}/moedas`] = aut.moedas + 3;
            if(a.tipo === "Trabalhar") up[`jogadores/${a.autorId}/moedas`] = aut.moedas + 1;
            if(a.tipo === "Propina") {
                up[`jogadores/${a.autorId}/moedas`] = aut.moedas - 4 + 3;
                up[`jogadores/${a.alvoId}/moedas`] = Math.max(0, d.jogadores[a.alvoId].moedas - 3);
            }
            if(a.tipo === "Golpe") {
                up[`jogadores/${a.autorId}/moedas`] = aut.moedas - 7;
                let cAlvo = d.jogadores[a.alvoId].cartas || [];
                if(cAlvo.length > 0) up[`jogadores/${a.alvoId}/cartas`] = cAlvo.slice(1);
            }
            up.fase = "ACAO"; up.acaoPendente = null;
            db.ref(`salas/${minhaInfo.sala}`).update(up).then(passarTurno);
        });
    }

    function mostrarCartaAoX9(idx) {
        db.ref(`salas/${minhaInfo.sala}`).once('value', s => {
            const d = s.val(); const carta = d.jogadores[minhaInfo.id].cartas[idx];
            alert(`Você mostrou ${carta} ao X9.`);
            db.ref(`salas/${minhaInfo.sala}`).update({ fase: "ACAO", acaoPendente: null }).then(passarTurno);
        });
    }

    function passarTurno() {
        db.ref(`salas/${minhaInfo.sala}`).once('value', s => {
            const d = s.val(); const ids = Object.keys(d.jogadores).filter(id => (d.jogadores[id].cartas || []).length > 0);
            const atualIdx = ids.indexOf(d.vezDe);
            const prox = ids[(atualIdx + 1) % ids.length];
            db.ref(`salas/${minhaInfo.sala}`).update({ vezDe: prox, fase: "ACAO", timerFinal: Date.now() + 30000, acaoPendente: null });
            alvoId = null;
        });
    }

    function perderCarta(id, cb) {
        db.ref(`salas/${minhaInfo.sala}/jogadores/${id}/cartas`).once('value', s => {
            let c = s.val() || []; c.shift();
            db.ref(`salas/${minhaInfo.sala}/jogadores/${id}/cartas`).set(c).then(cb);
        });
    }
