<script>
    const firebaseConfig = JSON.parse(__firebase_config);
    firebase.initializeApp(firebaseConfig);
    const db = firebase.database();

    let salaID, jogadorID, alvoID = null;
    let localCards = [];
    let localCoins = 0;
    let jogadoresCache = {};

    const TRADUCAO = {
        'trabalhar': 'Trampo', 'ajuda': 'Imposto', 'politico': 'Faz o L',
        'bicheiro': 'Bicho', 'investigar': 'X9', 'bandido': 'Vasco'
    };

    function entrar() {
        salaID = document.getElementById("input-sala").value.trim().toUpperCase();
        let nome = document.getElementById("input-nome").value.trim();
        if(!salaID || !nome) return;

        db.ref(`salas/${salaID}`).once('value', snap => {
            const data = snap.val();
            // Bloqueia entrada se o jogo já estiver rolando
            if(data && data.turnoAtual) {
                document.getElementById("login-status").innerText = "JOGO EM ANDAMENTO. ESPERE A PRÓXIMA!";
                document.getElementById("login-status").style.color = "var(--red)";
                return;
            }

            jogadorID = "p_" + Date.now();
            document.getElementById("display-nome").innerText = nome.toUpperCase();
            document.getElementById("screen-login").classList.remove("active");
            document.getElementById("screen-game").classList.add("active");

            const baralho = ["Político", "Político", "Miliciano", "Miliciano", "Bandido", "Bandido", "X9", "X9", "Juiz", "Juiz"];
            const c1 = baralho.splice(Math.floor(Math.random()*baralho.length), 1)[0];
            const c2 = baralho.splice(Math.floor(Math.random()*baralho.length), 1)[0];

            db.ref(`salas/${salaID}/jogadores/${jogadorID}`).set({
                nome, moedas: 1, vivo: true, cartas: [c1, c2]
            });

            ouvir();
        });
    }

    function ouvir() {
        db.ref(`salas/${salaID}`).on("value", snap => {
            const d = snap.val();
            if(!d) return;
            jogadoresCache = d.jogadores || {};
            render(d);
        });
    }

    function render(dados) {
        const eu = jogadoresCache[jogadorID];
        if(!eu) return;

        localCards = eu.cartas || [];
        localCoins = eu.moedas || 0;
        document.getElementById("my-coins").innerText = `💰 ${localCoins}`;
        document.getElementById("my-cards").innerHTML = localCards.map(c => `
            <div class="card-large"><i class="fa-solid ${getIcon(c)}"></i><span style="font-size:10px">${c.toUpperCase()}</span></div>
        `).join('');

        // Render Adversários
        const container = document.getElementById("adversarios-container");
        container.innerHTML = "";
        const outros = Object.keys(jogadoresCache).filter(id => id !== jogadorID);
        outros.forEach((id, i) => {
            const angle = (i / outros.length) * 2 * Math.PI - Math.PI/2;
            const x = Math.cos(angle) * 240;
            const y = Math.sin(angle) * 240;
            const isTurn = dados.turnoAtual === id;
            container.innerHTML += `
                <div class="player-node" style="left: calc(50% + ${x}px - 90px); top: calc(50% + ${y}px - 60px)">
                    <div class="node-box ${alvoID === id ? 'is-target' : ''} ${isTurn ? 'active-turn' : ''}" onclick="selecionarAlvo('${id}')">
                        <div class="node-name">${jogadoresCache[id].nome}</div>
                        <div class="node-coins">💰 ${jogadoresCache[id].moedas}</div>
                    </div>
                </div>
            `;
        });

        // Interface de Ações e Timers
        const acao = dados.acaoAtual;
        const acaoAtiva = acao && !acao.finalizada;
        const timerVal = document.getElementById("timer-val");
        const timerLabel = document.getElementById("timer-label");

        if(acaoAtiva) {
            const diff = Math.ceil((acao.tempoFim - Date.now()) / 1000);
            timerVal.innerText = Math.max(0, diff);
            timerLabel.innerText = "REAÇÃO";
            gestaoInterfaceReacao(acao);
        } else if(dados.turnoAtual) {
            const diffTurno = Math.ceil((dados.turnoFim - Date.now()) / 1000);
            timerVal.innerText = Math.max(0, diffTurno);
            timerLabel.innerText = "TURNO";
            document.getElementById("action-ui").style.display = "none";
        }

        // Bloqueios de Menu
        const isMyTurn = dados.turnoAtual === jogadorID;
        const menu = document.getElementById("meu-menu-acoes");
        const statusMsg = document.getElementById("status-turno-msg");
        const alerta10 = document.getElementById("alerta-10");

        if(isMyTurn && !acaoAtiva) {
            menu.style.opacity = "1";
            menu.style.pointerEvents = "all";
            statusMsg.style.display = "block";
            if(localCoins >= 10) {
                ["btn-trabalhar", "btn-ajuda", "btn-politico", "btn-bicheiro", "btn-investigar"].forEach(id => document.getElementById(id).disabled = true);
                alerta10.style.display = "block";
            } else {
                ["btn-trabalhar", "btn-ajuda", "btn-politico", "btn-bicheiro", "btn-investigar"].forEach(id => document.getElementById(id).disabled = false);
                alerta10.style.display = "none";
            }
        } else {
            menu.style.opacity = "0.3";
            menu.style.pointerEvents = "none";
            statusMsg.style.display = "none";
        }

        // X9 Privado para o Atacante
        if(dados.x9_result && dados.x9_result.para === jogadorID) {
            document.getElementById("x9-text").innerText = `${jogadoresCache[dados.x9_result.alvo].nome} mostrou a carta: ${dados.x9_result.carta}`;
            document.getElementById("x9-result-popup").style.display = "block";
            db.ref(`salas/${salaID}/x9_result`).remove();
        }

        // Logs
        const logBox = document.getElementById("log-box");
        if(dados.logs) {
            logBox.innerHTML = Object.values(dados.logs).reverse().map(l => `<div class="log-msg">${l}</div>`).join('');
        }
    }

    function selecionarAlvo(id) { alvoID = id; }

    function fazerAcao(tipo) {
        if(["bicheiro", "investigar", "bandido"].includes(tipo) && !alvoID) return alert("SELECIONE UM ALVO NA MESA!");

        // Custos imediatos
        if(tipo === 'bandido') {
            if(localCoins < 3) return alert("PRECISA DE 3 MOEDAS!");
            db.ref(`salas/${salaID}/jogadores/${jogadorID}/moedas`).transaction(m => Math.max(0, m - 3));
        }

        const acao = {
            tipo, de: jogadorID, para: alvoID,
            tempoFim: Date.now() + 15000, 
            finalizada: false,
            aceitos: {} // Lista de quem aceitou (clicou em ignorar)
        };

        db.ref(`salas/${salaID}/acaoAtual`).set(acao);
        db.ref(`salas/${salaID}/logs`).push(`🚩 <b>${jogadoresCache[jogadorID].nome}</b> tentou <b>${TRADUCAO[tipo]}</b>!`);
        
        if(tipo === 'trabalhar') db.ref(`salas/${salaID}/acaoAtual/finalizada`).set(true);
        alvoID = null;
    }

    function gestaoInterfaceReacao(acao) {
        const ui = document.getElementById("action-ui");
        const btns = document.getElementById("action-btns");
        const autorNome = jogadoresCache[acao.de].nome;
        const souAlvo = acao.para === jogadorID;
        const souAutor = acao.de === jogadorID;
        
        if(souAutor) { ui.style.display = "none"; return; }

        let mostrar = false;
        btns.innerHTML = "";

        // TELA UNIFICADA: Duvidar + Bloquear + Ignorar
        if(acao.tipo === 'politico' || acao.tipo === 'ajuda') {
            mostrar = true;
            document.getElementById("action-title").innerText = TRADUCAO[acao.tipo].toUpperCase();
            document.getElementById("action-desc").innerText = `${autorNome} está agindo. Você aceita ou contesta?`;
            
            btns.innerHTML += `<button class="btn-action btn-vasco" onclick="reagir('duvidar')">DUVIDAR</button>`;
            if(acao.tipo === 'ajuda') btns.innerHTML += `<button class="btn-action btn-block" onclick="reagir('bloquear', 'Político')">BLOQUEAR (POLÍTICO)</button>`;
            btns.innerHTML += `<button class="btn-action btn-pass" onclick="ignorarAcao()">IGNORAR (ACEITAR)</button>`;
        } else if(souAlvo) {
            mostrar = true;
            document.getElementById("action-title").innerText = TRADUCAO[acao.tipo].toUpperCase();
            document.getElementById("action-desc").innerText = `Você é o alvo de ${autorNome}. Defenda-se!`;
            
            btns.innerHTML += `<button class="btn-action btn-vasco" onclick="reagir('duvidar')">DUVIDAR</button>`;
            if(acao.tipo === 'bandido' || acao.tipo === 'bicheiro') btns.innerHTML += `<button class="btn-action btn-block" onclick="reagir('bloquear', 'Miliciano')">BLOQUEAR (MILICIANO)</button>`;
            if(acao.tipo === 'investigar') btns.innerHTML += `<button class="btn-action btn-block" onclick="reagir('bloquear', 'Juiz')">BLOQUEAR (JUIZ)</button>`;
            btns.innerHTML += `<button class="btn-action btn-pass" onclick="ignorarAcao()">IGNORAR (ACEITAR)</button>`;
        }

        if(mostrar) ui.style.display = "block";
    }

    function reagir(modo, personagem = "") {
        const autorID = jogadoresCache[jogadorID].acaoAtual_autor || ""; // Placeholder
        const piadas = ["O golpe tá aí, cai quem quer!", "Aqui não, violão!", "Tentou a sorte mas o azar é certo.", "Malandro demais o bicho come."];
        const piada = piadas[Math.floor(Math.random()*piadas.length)];

        db.ref(`salas/${salaID}/acaoAtual`).update({ finalizada: true, contestada: true });
        db.ref(`salas/${salaID}/logs`).push(`🔥 <b>${jogadoresCache[jogadorID].nome}</b> reagiu! <i>"${piada}"</i>`);
        document.getElementById("action-ui").style.display = "none";
    }

    function ignorarAcao() {
        db.ref(`salas/${salaID}/acaoAtual/aceitos/${jogadorID}`).set(true);
        document.getElementById("action-ui").style.display = "none";
    }

    function abrirSeletorCarta(titulo, modo, extras = {}) {
        const modal = document.getElementById("card-selector-modal");
        const container = document.getElementById("selector-cards-container");
        document.getElementById("selector-title").innerText = titulo;
        container.innerHTML = "";
        
        localCards.forEach((c, i) => {
            const div = document.createElement("div");
            div.className = "card-large";
            div.innerHTML = `<i class="fa-solid ${getIcon(c)}"></i><span>${c}</span>`;
            div.onclick = () => {
                if(modo === "perder") {
                    let n = [...localCards]; n.splice(i, 1);
                    db.ref(`salas/${salaID}/jogadores/${jogadorID}/cartas`).set(n);
                } else if(modo === "x9_show") {
                    db.ref(`salas/${salaID}/x9_result`).set({ para: extras.para, alvo: jogadorID, carta: c });
                }
                modal.style.display = "none";
            };
            container.appendChild(div);
        });
        modal.style.display = "flex";
    }

    function resetarJogo() {
        if(confirm("LIMPAR A SALA E REINICIAR TUDO?")) {
            db.ref(`salas/${salaID}`).remove();
            location.reload();
        }
    }

    // SIMULAÇÃO DE SERVIDOR
    setInterval(() => {
        db.ref(`salas/${salaID}`).once('value', snap => {
            const d = snap.val();
            if(!d || !d.jogadores) return;

            // Logica do Politico (Todos aceitarem)
            if(d.acaoAtual && !d.acaoAtual.finalizada) {
                const acao = d.acaoAtual;
                if(acao.tipo === 'politico' || acao.tipo === 'ajuda') {
                    const aceitosCount = Object.keys(acao.aceitos || {}).length;
                    const totalAdversarios = Object.keys(d.jogadores).length - 1;
                    if(aceitosCount >= totalAdversarios) concluirAcao(acao, d);
                }
                // Se passou o tempo
                if(Date.now() > acao.tempoFim) concluirAcao(acao, d);
                return;
            }

            // Inicia Turno
            if(!d.turnoFim) {
                db.ref(`salas/${salaID}`).update({
                    turnoAtual: Object.keys(d.jogadores)[0],
                    turnoFim: Date.now() + 30000
                });
                return;
            }

            // Próximo Turno
            if(Date.now() > d.turnoFim) passarTurno(d);
        });
    }, 1500);

    function concluirAcao(acao, dados) {
        if(acao.contestada) { 
            db.ref(`salas/${salaID}/acaoAtual/finalizada`).set(true); 
            return; 
        }

        const deRef = db.ref(`salas/${salaID}/jogadores/${acao.de}`);
        if(acao.tipo === 'trabalhar') deRef.child("moedas").transaction(m => Math.min(10, (m || 0) + 1));
        if(acao.tipo === 'ajuda') deRef.child("moedas").transaction(m => Math.min(10, (m || 0) + 2));
        if(acao.tipo === 'politico') deRef.child("moedas").transaction(m => Math.min(10, (m || 0) + 3));
        
        if(acao.tipo === 'bandido' && acao.para === jogadorID) abrirSeletorCarta("PERDER UMA CARTA", "perder");
        if(acao.tipo === 'investigar' && acao.para === jogadorID) abrirSeletorCarta("X9: QUAL CARTA MOSTRAR?", "x9_show", { para: acao.de });

        db.ref(`salas/${salaID}/acaoAtual/finalizada`).set(true);
    }

    function passarTurno(dados) {
        const jids = Object.keys(dados.jogadores);
        const cur = jids.indexOf(dados.turnoAtual);
        const next = jids[(cur + 1) % jids.length];
        db.ref(`salas/${salaID}`).update({
            turnoAtual: next, turnoFim: Date.now() + 30000, acaoAtual: null
        });
    }

        return icons[c] || "fa-id-card";
