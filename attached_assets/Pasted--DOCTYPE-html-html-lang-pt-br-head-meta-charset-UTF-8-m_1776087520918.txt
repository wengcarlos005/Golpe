<script>
    const firebaseConfig = { apiKey: "AIzaSyCj6Q8eWBiiBXOx_7vOICzGHjBSDYFNV2Q", databaseURL: "https://golpe-63ee3-default-rtdb.firebaseio.com", projectId: "golpe-63ee3" };
    firebase.initializeApp(firebaseConfig);
    const db = firebase.database();

    let minhaInfo = { nome: "", id: "", sala: "", isHost: false };
    let alvoId = null;

    function nav(id) { document.querySelectorAll('.screen').forEach(s => s.classList.remove('active')); document.getElementById(id).classList.add('active'); }

    function criarSala() {
        const n = document.getElementById('input-nome').value; if(!n) return;
        const cod = Math.random().toString(36).substring(2, 6).toUpperCase();
        minhaInfo = { nome: n, id: "p_"+Date.now(), sala: cod, isHost: true };
        db.ref(`salas/${cod}`).set({ iniciado: false, host: minhaInfo.id, fase: "ESPERA", log: [] });
        conectar();
    }

    function entrarSala() {
        const n = document.getElementById('input-nome').value;
        const cod = document.getElementById('input-sala-cod').value.toUpperCase();
        if(!n || !cod) return;
        minhaInfo = { nome: n, id: "p_"+Date.now(), sala: cod, isHost: false };
        conectar();
    }

    function conectar() {
        db.ref(`salas/${minhaInfo.sala}/jogadores/${minhaInfo.id}`).set({ nome: minhaInfo.nome, moedas: 2, vivo: true, host: minhaInfo.isHost });
        document.getElementById('display-cod').innerText = minhaInfo.sala;
        nav('screen-lobby');
        ouvirSala();
    }

    function ouvirSala() {
        db.ref(`salas/${minhaInfo.sala}`).on('value', snap => {
            const d = snap.val(); if(!d) return;
            if(d.iniciado && document.getElementById('screen-lobby').classList.contains('active')) nav('screen-mesa');
            renderLobby(d);
            if(d.iniciado) renderMesa(d);
        });
    }

    function renderLobby(d) {
        const jogs = d.jogadores || {};
        const lista = document.getElementById('lista-jogadores');
        lista.innerHTML = "";
        Object.values(jogs).forEach(j => { lista.innerHTML += `<div>• ${j.nome}</div>`; });
        if(minhaInfo.isHost) {
            const btn = document.getElementById('btn-iniciar');
            btn.disabled = Object.keys(jogs).length < 2;
            btn.innerText = "INICIAR PARTIDA";
        }
    }

    function iniciarPartida() {
        let b = []; ["Político","Bicheiro","Bandido","X9","Juiz","Miliciano"].forEach(p => { for(let i=0; i<3; i++) b.push(p); });
        b.sort(() => Math.random() - 0.5);
        db.ref(`salas/${minhaInfo.sala}/jogadores`).once("value", s => {
            const ids = Object.keys(s.val());
            let up = { iniciado: true, vezDe: ids[0], fase: "ACAO", timerFinal: Date.now() + 30000, baralho: b };
            ids.forEach(id => { up[`jogadores/${id}/cartas`] = [b.pop(), b.pop()]; up[`jogadores/${id}/moedas`] = 2; });
            db.ref(`salas/${minhaInfo.sala}`).update(up);
        });
    }

    function renderMesa(d) {
        const jogs = d.jogadores || {};
        const ativos = Object.keys(jogs).filter(id => (jogs[id].cartas || []).length > 0);
        if(ativos.length === 1) { document.getElementById('area-botoes').innerHTML = `🏆 VITORIA: ${jogs[ativos[0]].nome}`; return; }

        let advH = "";
        for(let id in jogs) {
            if(id === minhaInfo.id) continue;
            const vivo = (jogs[id].cartas || []).length > 0;
            let st = (id === d.vezDe ? "active-turn" : (id === alvoId ? "target-selected" : ""));
            advH += `<div class="player-card ${st}" onclick="${vivo ? `alvoId='${id}'` : ''}">${jogs[id].nome}<br>${vivo ? '💰'+jogs[id].moedas : '💀'}</div>`;
        }
        document.getElementById('adversarios').innerHTML = advH;

        const eu = jogs[minhaInfo.id];
        document.getElementById('minhas-moedas').innerText = `💰 ${eu.moedas}`;
        document.getElementById('minhas-cartas').innerHTML = (eu.cartas || []).map(c => `<div class="minha-carta">${c}</div>`).join("");
        
        const area = document.getElementById('area-botoes'); area.innerHTML = "";
        if((eu.cartas || []).length === 0) return;

        if(d.fase === "ACAO" && d.vezDe === minhaInfo.id) {
            area.innerHTML += `<button class="btn-yellow" onclick="pedirAcao('Político')">Político (+3💰)</button>`;
            area.innerHTML += `<button class="btn-yellow" onclick="pedirAcao('Ajuda Externa')">Ajuda Externa (+2💰)</button>`;
            area.innerHTML += `<button class="btn-yellow" onclick="pedirAcao('Bicheiro')">Roubar (Bicheiro)</button>`;
            if(eu.moedas >= 3) area.innerHTML += `<button class="btn-yellow" onclick="pedirAcao('Bandido')" style="background:#d32f2f; color:#fff">Assassinar (Bandido)</button>`;
            area.innerHTML += `<button class="btn-yellow" onclick="pedirAcao('Trabalhar')" style="background:#444; color:#fff">Trabalhar (+1💰)</button>`;
            if(eu.moedas >= 7) area.innerHTML += `<button class="btn-yellow" onclick="pedirAcao('Golpe')" style="background:red; color:#fff">GOLPE (-7💰)</button>`;
        } else if(d.fase === "DUVIDA" && d.acaoPendente.autorId !== minhaInfo.id) {
            area.innerHTML += `<button class="btn-yellow" onclick="reagir('DUVIDO')">DUVIDAR</button>`;
            area.innerHTML += `<button class="btn-yellow" onclick="reagir('PASSAR')">PASSAR</button>`;
        } else if(d.fase === "BLOQUEIO" && d.acaoPendente.alvoId === minhaInfo.id) {
            area.innerHTML += `<button class="btn-yellow" onclick="reagir('BLOQUEIO')">BLOQUEAR</button>`;
            area.innerHTML += `<button class="btn-yellow" onclick="reagir('PASSAR')">ACEITAR</button>`;
        }
    }

    function pedirAcao(tipo) {
        if(["Bicheiro","Bandido","Golpe"].includes(tipo) && !alvoId) return alert("Selecione um alvo!");
        let f = (tipo === "Trabalhar" || tipo === "Golpe") ? "EXECUCAO" : "DUVIDA";
        db.ref(`salas/${minhaInfo.sala}`).update({ acaoPendente: { autorId: minhaInfo.id, tipo, alvoId: alvoId || null }, fase: f, timerFinal: Date.now() + 10000 });
        if(f === "EXECUCAO") executarEfeito();
    }

    function reagir(r) {
        db.ref(`salas/${minhaInfo.sala}`).once('value', s => {
            const d = s.val(); const a = d.acaoPendente;
            if(r === "DUVIDO") {
                let tem = d.jogadores[a.autorId].cartas.includes(a.tipo);
                tem ? perderCarta(minhaInfo.id, executarEfeito) : perderCarta(a.autorId, passarTurno);
            } else if(r === "BLOQUEIO") {
                db.ref(`salas/${minhaInfo.sala}`).update({ fase: "DUVIDA_BLOQUEIO", timerFinal: Date.now() + 10000 });
            } else {
                if(d.fase === "DUVIDA" && ["Bicheiro","Bandido","Ajuda Externa"].includes(a.tipo)) {
                    db.ref(`salas/${minhaInfo.sala}`).update({ fase: "BLOQUEIO", timerFinal: Date.now() + 10000 });
                } else { executarEfeito(); }
            }
        });
    }

    function executarEfeito() {
        db.ref(`salas/${minhaInfo.sala}`).once('value', s => {
            const d = s.val(); const a = d.acaoPendente; if(!a) return;
            let up = {}; const aut = d.jogadores[a.autorId];
            if(a.tipo === "Político") up[`jogadores/${a.autorId}/moedas`] = aut.moedas + 3;
            if(a.tipo === "Ajuda Externa") up[`jogadores/${a.autorId}/moedas`] = aut.moedas + 2;
            if(a.tipo === "Trabalhar") up[`jogadores/${a.autorId}/moedas`] = aut.moedas + 1;
            if(a.tipo === "Bicheiro") {
                up[`jogadores/${a.autorId}/moedas`] = aut.moedas + 2;
                up[`jogadores/${a.alvoId}/moedas`] = Math.max(0, d.jogadores[a.alvoId].moedas - 2);
            }
            if(a.tipo === "Bandido") {
                up[`jogadores/${a.autorId}/moedas`] = aut.moedas - 3;
                perderCarta(a.alvoId, () => {});
            }
            if(a.tipo === "Golpe") {
                up[`jogadores/${a.autorId}/moedas`] = aut.moedas - 7;
                perderCarta(a.alvoId, () => {});
            }
            up.fase = "ACAO"; up.acaoPendente = null;
            db.ref(`salas/${minhaInfo.sala}`).update(up).then(passarTurno);
        });
    }

    function passarTurno() {
        db.ref(`salas/${minhaInfo.sala}`).once('value', s => {
            const d = s.val(); const ids = Object.keys(d.jogadores).filter(id => (d.jogadores[id].cartas || []).length > 0);
            const prox = ids[(ids.indexOf(d.vezDe) + 1) % ids.length];
            db.ref(`salas/${minhaInfo.sala}`).update({ vezDe: prox, fase: "ACAO", timerFinal: Date.now() + 30000, acaoPendente: null });
            alvoId = null;
        });
    }

    function perderCarta(id, cb) {
        db.ref(`salas/${minhaInfo.sala}/jogadores/${id}/cartas`).once('value', s => {
            let c = s.val() || []; c.shift();
            db.ref(`salas/${minhaInfo.sala}/jogadores/${id}/cartas`).set(c).then(cb);
        });
    }
