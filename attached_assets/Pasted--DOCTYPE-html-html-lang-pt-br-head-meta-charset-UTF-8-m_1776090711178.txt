<script>
    const firebaseConfig = { apiKey: "AIzaSyCj6Q8eWBiiBXOx_7vOICzGHjBSDYFNV2Q", databaseURL: "https://golpe-63ee3-default-rtdb.firebaseio.com", projectId: "golpe-63ee3" };
    firebase.initializeApp(firebaseConfig);
    const db = firebase.database();

    let minhaInfo = { nome: "", id: "", sala: "", host: false };
    let alvosSel = [];

    function nav(id) { document.querySelectorAll('.screen').forEach(s => s.classList.remove('active')); document.getElementById(id).classList.add('active'); }

    function criarSala() {
        const cod = Math.random().toString(36).substring(2,6).toUpperCase();
        minhaInfo = { nome: document.getElementById('input-nome').value, id: "p"+Date.now(), sala: cod, host: true };
        db.ref(`salas/${cod}`).set({ iniciado: false, fase: "ESPERA" });
        entrar();
    }

    function entrarSala() {
        minhaInfo = { nome: document.getElementById('input-nome').value, id: "p"+Date.now(), sala: document.getElementById('input-sala-cod').value.toUpperCase(), host: false };
        entrar();
    }

    function entrar() {
        db.ref(`salas/${minhaInfo.sala}/jogadores/${minhaInfo.id}`).set({ nome: minhaInfo.nome, moedas: 2, vivo: true });
        document.getElementById('display-cod').innerText = minhaInfo.sala;
        nav('screen-lobby');
        ouvir();
    }

    function ouvir() {
        db.ref(`salas/${minhaInfo.sala}`).on('value', snap => {
            const d = snap.val(); if(!d) return;
            if(d.iniciado) { nav('screen-mesa'); renderMesa(d); }
            else renderLobby(d);
        });
    }

    function renderLobby(d) {
        const lista = document.getElementById('lista-jogadores'); lista.innerHTML = "";
        const jogs = Object.values(d.jogadores || {});
        jogs.forEach(j => lista.innerHTML += `<div>• ${j.nome}</div>`);
        if(minhaInfo.host) document.getElementById('btn-iniciar').disabled = jogs.length < 2;
    }

    function iniciarPartida() {
        let b = []; ["Político","Bicheiro","Bandido","X9","Juiz","Miliciano"].forEach(p => { for(let i=0; i<3; i++) b.push(p); });
        b.sort(() => Math.random() - 0.5);
        db.ref(`salas/${minhaInfo.sala}/jogadores`).once('value', s => {
            const ids = Object.keys(s.val());
            let up = { iniciado: true, vezDe: ids[0], fase: "ACAO", baralho: b };
            ids.forEach(id => { up[`jogadores/${id}/cartas`] = [b.pop(), b.pop()]; });
            db.ref(`salas/${minhaInfo.sala}`).update(up);
        });
    }

    function renderMesa(d) {
        const js = d.jogadores; const eu = js[minhaInfo.id];
        const area = document.getElementById('area-botoes'); area.innerHTML = "";
        const mCards = document.getElementById('minhas-cartas'); mCards.innerHTML = "";
        const advs = document.getElementById('adversarios'); advs.innerHTML = "";

        document.getElementById('minhas-moedas').innerText = `💰 ${eu.moedas}`;
        document.getElementById('status-fase').innerText = d.fase;
        
        (eu.cartas || []).forEach(c => { mCards.innerHTML += `<div class="minha-carta">${c}</div>`; });

        const vivos = Object.keys(js).filter(id => (js[id].cartas || []).length > 0);
        vivos.forEach(id => {
            if(id === minhaInfo.id) return;
            let sel = alvosSel.includes(id) ? "selected" : "";
            advs.innerHTML += `<div class="player-card ${sel} ${id === d.vezDe ? 'active-turn' : ''}" onclick="toggleAlvo('${id}', ${vivos.length})">${js[id].nome}<br>💰${js[id].moedas}</div>`;
        });

        if(d.fase === "ACAO" && d.vezDe === minhaInfo.id) {
            area.innerHTML += `<button class="btn-yellow" onclick="agir('Político')">Político (+3💰)</button>`;
            area.innerHTML += `<button class="btn-yellow" onclick="agir('Ajuda Externa')">Ajuda Externa (+2💰)</button>`;
            area.innerHTML += `<button class="btn-yellow" onclick="agir('Bicheiro')">Bicheiro (Roubar)</button>`;
            area.innerHTML += `<button class="btn-yellow" onclick="agir('Investigar')">X9 (Investigar)</button>`;
            area.innerHTML += `<button class="btn-yellow" onclick="agir('Trocar')">X9: Trocar (1💰)</button>`;
            area.innerHTML += `<button class="btn-yellow" onclick="agir('Trabalhar')">Trabalhar (+1💰)</button>`;
        } else if((d.fase === "DUVIDA" || d.fase === "BLOQUEIO") && d.acaoPendente.autorId !== minhaInfo.id) {
            const votou = d.votos && d.votos[minhaInfo.id];
            if(!votou) {
                if(d.fase === "DUVIDA") area.innerHTML += `<button class="btn-yellow" onclick="votar('DUVIDAR')">DUVIDAR</button>`;
                area.innerHTML += `<button class="btn-yellow" onclick="votar('PASSAR')">PASSAR</button>`;
                
                if(d.fase === "BLOQUEIO" && d.acaoPendente.alvoId === minhaInfo.id) {
                    let btnTxt = "Bloquear";
                    if(d.acaoPendente.tipo === "Ajuda Externa") btnTxt = "Bloquear com Político";
                    if(["Bicheiro","Bandido"].includes(d.acaoPendente.tipo)) btnTxt = "Bloquear com Miliciano";
                    if(["Investigar","Trocar"].includes(d.acaoPendente.tipo)) btnTxt = "Bloquear com Juiz";
                    area.innerHTML += `<button class="btn-yellow" onclick="bloquear('${btnTxt.split(' ').pop()}')">${btnTxt}</button>`;
                }
            }
        }
        
        if(d.votos) {
            const count = Object.keys(d.votos).length;
            document.getElementById('votos-info').innerText = `Confirmados: ${count}/${vivos.length - 1}`;
        }
    }

    function toggleAlvo(id, totalVivos) {
        const max = totalVivos > 2 ? 2 : 1;
        if(alvosSel.includes(id)) alvosSel = alvosSel.filter(x => x !== id);
        else if(alvosSel.length < max) alvosSel.push(id);
        db.ref(`salas/${minhaInfo.sala}/ping`).set(Date.now()); // Força refresh local
    }

    function agir(tipo) {
        db.ref(`salas/${minhaInfo.sala}/jogadores`).once('value', s => {
            const vivos = Object.keys(s.val()).filter(id => (s.val()[id].cartas || []).length > 0);
            if(tipo === "Trocar") {
                if(vivos.length > 2 && alvosSel.length !== 2) return alert("Selecione 2 alvos!");
                if(vivos.length === 2 && alvosSel.length !== 1) return alert("Selecione o oponente!");
            }
            let f = (tipo === "Trabalhar") ? "EXECUCAO" : "DUVIDA";
            db.ref(`salas/${minhaInfo.sala}`).update({ 
                fase: f, 
                acaoPendente: { autorId: minhaInfo.id, tipo, alvos: alvosSel, alvoId: alvosSel[0]||null },
                votos: {} 
            });
            alvosSel = [];
        });
    }

    function votar(v) {
        if(v === "DUVIDAR") { /* Lógica de desafio... */ }
        else {
            db.ref(`salas/${minhaInfo.sala}/votos/${minhaInfo.id}`).set(true).then(checarConsenso);
        }
    }

    function checarConsenso() {
        db.ref(`salas/${minhaInfo.sala}`).once('value', s => {
            const d = s.val();
            const vivos = Object.keys(d.jogadores).filter(id => (d.jogadores[id].cartas || []).length > 0);
            const votos = Object.keys(d.votos || {}).length;
            if(votos >= vivos.length - 1) {
                if(d.fase === "DUVIDA") db.ref(`salas/${minhaInfo.sala}`).update({ fase: "BLOQUEIO", votos: {} });
                else finalizarAcao();
            }
        });
    }

    function finalizarAcao() {
        db.ref(`salas/${minhaInfo.sala}`).once('value', s => {
            const d = s.val(); const a = d.acaoPendente;
            let up = { fase: "ACAO", acaoPendente: null, votos: {} };
            const autor = d.jogadores[a.autorId];
            
            if(a.tipo === "Político") up[`jogadores/${a.autorId}/moedas`] = autor.moedas + 3;
            if(a.tipo === "Ajuda Externa") up[`jogadores/${a.autorId}/moedas`] = autor.moedas + 2;
            if(a.tipo === "Trocar") {
                up[`jogadores/${a.autorId}/moedas`] = autor.moedas - 1;
                // Se 2 jogadores, troca alvo com baralho. Se 3+, troca entre alvos.
                if(Object.keys(d.jogadores).length === 2) {
                    /* Lógica baralho... */
                }
            }
            db.ref(`salas/${minhaInfo.sala}`).update(up).then(passarTurno);
        });
    }

    function passarTurno() {
        db.ref(`salas/${minhaInfo.sala}`).once('value', s => {
            const d = s.val();
            const ids = Object.keys(d.jogadores).filter(id => (d.jogadores[id].cartas || []).length > 0);
            const prox = ids[(ids.indexOf(d.vezDe) + 1) % ids.length];
            db.ref(`salas/${minhaInfo.sala}`).update({ vezDe: prox, fase: "ACAO" });
        });
    }
