<script>
    const firebaseConfig = { apiKey: "AIzaSyCj6Q8eWBiiBXOx_7vOICzGHjBSDYFNV2Q", databaseURL: "https://golpe-63ee3-default-rtdb.firebaseio.com", projectId: "golpe-63ee3" };
    firebase.initializeApp(firebaseConfig);
    const db = firebase.database();

    let minhaInfo = { nome: "", id: "", sala: "", isHost: false };
    let alvosX9 = []; 

    function setDesc(t, d) { document.getElementById('desc-box').innerHTML = `<strong>${t}</strong>: ${d}`; }
    function nav(id) { document.querySelectorAll('.screen').forEach(s => s.classList.remove('active')); document.getElementById(id).classList.add('active'); }

    function criarSala() {
        const n = document.getElementById('input-nome').value; if(!n) return;
        const cod = Math.random().toString(36).substring(2, 6).toUpperCase();
        minhaInfo = { nome: n, id: "p_"+Date.now(), sala: cod, isHost: true };
        db.ref(`salas/${cod}`).set({ iniciado: false, fase: "ESPERA", log: ["Sala criada."] });
        conectar();
    }

    function entrarSala() {
        const n = document.getElementById('input-nome').value;
        const cod = document.getElementById('input-sala-cod').value.toUpperCase();
        if(!n || !cod) return;
        minhaInfo = { nome: n, id: "p_"+Date.now(), sala: cod, isHost: false };
        conectar();
    }

    function conectar() {
        db.ref(`salas/${minhaInfo.sala}/jogadores/${minhaInfo.id}`).set({ nome: minhaInfo.nome, moedas: 2, vivo: true });
        document.getElementById('display-cod').innerText = minhaInfo.sala;
        nav('screen-lobby');
        ouvirSala();
    }

    function ouvirSala() {
        db.ref(`salas/${minhaInfo.sala}`).on('value', snap => {
            const d = snap.val(); if(!d) return;
            if(d.iniciado && document.getElementById('screen-lobby').classList.contains('active')) nav('screen-mesa');
            renderLobby(d);
            if(d.iniciado) renderMesa(d);
        });
    }

    function renderLobby(d) {
        const jogs = d.jogadores || {};
        const lista = document.getElementById('lista-jogadores'); lista.innerHTML = "";
        Object.values(jogs).forEach(j => { lista.innerHTML += `<div>• ${j.nome}</div>`; });
        if(minhaInfo.isHost) {
            const btn = document.getElementById('btn-iniciar');
            btn.disabled = Object.keys(jogs).length < 2;
            btn.innerText = "INICIAR PARTIDA";
        }
    }

    function iniciarPartida() {
        let b = []; ["Político","Bicheiro","Bandido","X9","Juiz","Miliciano"].forEach(p => { for(let i=0; i<3; i++) b.push(p); });
        b.sort(() => Math.random() - 0.5);
        db.ref(`salas/${minhaInfo.sala}/jogadores`).once("value", s => {
            const ids = Object.keys(s.val());
            let up = { iniciado: true, vezDe: ids[0], fase: "ACAO", baralho: b };
            ids.forEach(id => { up[`jogadores/${id}/cartas`] = [b.pop(), b.pop()]; up[`jogadores/${id}/moedas`] = 2; });
            db.ref(`salas/${minhaInfo.sala}`).update(up);
        });
    }

    function renderMesa(d) {
        const jogs = d.jogadores || {};
        const eu = jogs[minhaInfo.id];
        const area = document.getElementById('area-botoes'); area.innerHTML = "";
        const mCards = document.getElementById('minhas-cartas'); mCards.innerHTML = "";
        document.getElementById('minhas-moedas').innerText = `💰 ${eu.moedas}`;
        
        let advH = "";
        for(let id in jogs) {
            if(id === minhaInfo.id) continue;
            const vivo = (jogs[id].cartas || []).length > 0;
            let sel = alvosX9.includes(id) ? "target-selected" : "";
            advH += `<div class="player-card ${id === d.vezDe ? 'active-turn' : ''} ${sel}" onclick="selecionarAlvo('${id}')">${jogs[id].nome}<br>${vivo ? '💰'+jogs[id].moedas : '💀'}</div>`;
        }
        document.getElementById('adversarios').innerHTML = advH;

        if(d.fase === "PERDER_CARTA" && d.quemPerde === minhaInfo.id) {
            document.getElementById('status-fase').innerText = "ESCOLHA A CARTA PARA DESCARTAR";
            (eu.cartas || []).forEach((c, i) => { mCards.innerHTML += `<div class="minha-carta" style="border:2px solid red" onclick="confirmarPerda(${i})">${c}</div>`; });
            return;
        }

        (eu.cartas || []).forEach(c => { mCards.innerHTML += `<div class="minha-carta">${c}</div>`; });
        document.getElementById('status-fase').innerText = d.fase;

        if(d.fase === "ACAO" && d.vezDe === minhaInfo.id) {
            area.innerHTML += `<button class="btn-yellow" onclick="pedirAcao('Político')">Político (+3💰)</button>`;
            area.innerHTML += `<button class="btn-yellow" onclick="pedirAcao('Ajuda Externa')">Ajuda Externa (+2💰)</button>`;
            area.innerHTML += `<button class="btn-yellow" onclick="pedirAcao('Bicheiro')">Bicheiro (Roubar)</button>`;
            if(eu.moedas >= 3) area.innerHTML += `<button class="btn-yellow" onclick="pedirAcao('Bandido')" style="color:red">Bandido (Assassinar)</button>`;
            area.innerHTML += `<button class="btn-yellow" onclick="pedirAcao('Investigar')">X9 (Investigar)</button>`;
            area.innerHTML += `<button class="btn-yellow" onclick="pedirAcao('Trocar')">X9 (Trocar Cartas)</button>`;
            area.innerHTML += `<button class="btn-yellow" onclick="pedirAcao('Disfarce')">X9 (Disfarce)</button>`;
            area.innerHTML += `<button class="btn-yellow" onclick="pedirAcao('Trabalhar')">Trabalhar (+1💰)</button>`;
        } else if(d.fase === "DUVIDA" && d.acaoPendente.autorId !== minhaInfo.id) {
            area.innerHTML += `<button class="btn-yellow" onclick="reagir('DUVIDO')">DUVIDAR</button>`;
            area.innerHTML += `<button class="btn-yellow" onclick="reagir('PASSAR')">PASSAR</button>`;
        } else if(d.fase === "BLOQUEIO" && d.acaoPendente.alvoId === minhaInfo.id) {
            const a = d.acaoPendente.tipo;
            if(a === "Ajuda Externa") area.innerHTML += `<button class="btn-yellow" onclick="reagir('BLOQUEIO','Político')">Bloquear com Político</button>`;
            if(a === "Bicheiro" || a === "Bandido") area.innerHTML += `<button class="btn-yellow" onclick="reagir('BLOQUEIO','Miliciano')">Bloquear com Miliciano</button>`;
            if(["Investigar","Trocar","Disfarce"].includes(a)) area.innerHTML += `<button class="btn-yellow" onclick="reagir('BLOQUEIO','Juiz')">Bloquear com Juiz</button>`;
            area.innerHTML += `<button class="btn-yellow" onclick="reagir('PASSAR')">ACEITAR</button>`;
        } else if(d.fase === "DUVIDA_BLOQUEIO" && d.acaoPendente.autorId === minhaInfo.id) {
            area.innerHTML += `<p>Duvidar do Bloqueio de ${d.bloqueadorPersonagem}?</p>`;
            area.innerHTML += `<button class="btn-yellow" onclick="contestarBloqueio(true)">DUVIDAR DO BLOQUEIO</button>`;
            area.innerHTML += `<button class="btn-yellow" onclick="contestarBloqueio(false)">ACEITAR BLOQUEIO</button>`;
        }
    }

    function selecionarAlvo(id) {
        if(alvosX9.includes(id)) alvosX9 = alvosX9.filter(x => x !== id);
        else if(alvosX9.length < 2) alvosX9.push(id);
    }

    function pedirAcao(tipo) {
        const needsAlvo = ["Bicheiro","Bandido","Investigar","Trocar"];
        if(needsAlvo.includes(tipo) && alvosX9.length === 0) return alert("Selecione o alvo!");
        let f = (tipo === "Trabalhar" || tipo === "Ajuda Externa") ? "BLOQUEIO" : "DUVIDA";
        if(tipo === "Trabalhar") f = "EXECUCAO";
        db.ref(`salas/${minhaInfo.sala}`).update({ acaoPendente: { autorId: minhaInfo.id, tipo, alvos: alvosX9, alvoId: alvosX9[0]||null }, fase: f });
        if(f === "EXECUCAO") executarEfeito();
        alvosX9 = [];
    }

    function reagir(r, char) {
        db.ref(`salas/${minhaInfo.sala}`).once('value', s => {
            const d = s.val(); const a = d.acaoPendente;
            if(r === "DUVIDO") {
                let realChar = (["Investigar","Trocar","Disfarce"].includes(a.tipo)) ? "X9" : a.tipo;
                let tem = d.jogadores[a.autorId].cartas.includes(realChar);
                tem ? entrarFasePerda(minhaInfo.id) : entrarFasePerda(a.autorId);
            } else if(r === "BLOQUEIO") {
                db.ref(`salas/${minhaInfo.sala}`).update({ fase: "DUVIDA_BLOQUEIO", bloqueadorId: minhaInfo.id, bloqueadorPersonagem: char });
            } else {
                if(d.fase === "DUVIDA") db.ref(`salas/${minhaInfo.sala}`).update({ fase: "BLOQUEIO" });
                else executarEfeito();
            }
        });
    }

    function contestarBloqueio(duvida) {
        db.ref(`salas/${minhaInfo.sala}`).once('value', s => {
            const d = s.val();
            if(!duvida) return passarTurno();
            let tem = d.jogadores[d.bloqueadorId].cartas.includes(d.bloqueadorPersonagem);
            if(tem) {
                // Bloqueador estava certo: Autor perde as moedas (se bandido/ajuda) e perde carta
                entrarFasePerda(minhaInfo.id);
            } else {
                // Bloqueador mentiu: Bloqueador perde carta e ação executa
                entrarFasePerda(d.bloqueadorId, true);
            }
        });
    }

    function executarEfeito() {
        db.ref(`salas/${minhaInfo.sala}`).once('value', s => {
            const d = s.val(); const a = d.acaoPendente; if(!a) return;
            let up = {}; const aut = d.jogadores[a.autorId];
            if(a.tipo === "Político") up[`jogadores/${a.autorId}/moedas`] = aut.moedas + 3;
            if(a.tipo === "Ajuda Externa") up[`jogadores/${a.autorId}/moedas`] = aut.moedas + 2;
            if(a.tipo === "Bicheiro") { up[`jogadores/${a.autorId}/moedas`] = aut.moedas + 2; up[`jogadores/${a.alvoId}/moedas`] = Math.max(0, d.jogadores[a.alvoId].moedas - 2); }
            if(a.tipo === "Bandido") { up[`jogadores/${a.autorId}/moedas`] = Math.max(0, aut.moedas - 3); db.ref(`salas/${minhaInfo.sala}`).update({ fase: "PERDER_CARTA", quemPerde: a.alvoId }); return; }
            if(a.tipo === "Investigar") { alert("Você viu: " + d.jogadores[a.alvoId].cartas[0]); }
            if(a.tipo === "Trocar" && a.alvos.length === 2) {
                let c1 = d.jogadores[a.alvos[0]].cartas; let c2 = d.jogadores[a.alvos[1]].cartas;
                up[`jogadores/${a.alvos[0]}/cartas`] = [c2.shift(), ...c1]; up[`jogadores/${a.alvos[1]}/cartas`] = [c1.shift(), ...c2];
            }
            up.fase = "ACAO"; up.acaoPendente = null;
            db.ref(`salas/${minhaInfo.sala}`).update(up).then(passarTurno);
        });
    }

    function entrarFasePerda(id, e d) { db.ref(`salas/${minhaInfo.sala}`).update({ fase: "PERDER_CARTA", quemPerde: id }); }
    function confirmarPerda(idx) {
        db.ref(`salas/${minhaInfo.sala}/jogadores/${minhaInfo.id}/cartas`).once('value', s => {
            let c = s.val() || []; c.splice(idx, 1);
            db.ref(`salas/${minhaInfo.sala}/jogadores/${minhaInfo.id}/cartas`).set(c).then(() => {
                db.ref(`salas/${minhaInfo.sala}`).update({ fase: "ACAO" }).then(passarTurno);
            });
        });
    }
    function passarTurno() {
        db.ref(`salas/${minhaInfo.sala}`).once('value', s => {
            const d = s.val(); const ids = Object.keys(d.jogadores).filter(id => (d.jogadores[id].cartas || []).length > 0);
            const prox = ids[(ids.indexOf(d.vezDe) + 1) % ids.length];
            db.ref(`salas/${minhaInfo.sala}`).update({ vezDe: prox, fase: "ACAO", acaoPendente: null });
        });
    }
