<script>
    const firebaseConfig = {
        apiKey: "AIzaSyCj6Q8eWBiiBXOx_7vOICzGHjBSDYFNV2Q",
        databaseURL: "https://golpe-63ee3-default-rtdb.firebaseio.com",
        projectId: "golpe-63ee3",
    };
    firebase.initializeApp(firebaseConfig);
    const db = firebase.database();

    let salaID, jogadorID, alvoID = null;
    let localCards = [];
    let localCoins = 0;

    const TRADUCAO = {
        'trabalhar': 'Trampo', 'ajuda': 'Imposto', 'politico': 'Faz o L',
        'bicheiro': 'Bicho', 'investigar': 'X9', 'assassino': 'Vasco'
    };

    function entrar() {
        salaID = document.getElementById("input-sala").value;
        let nome = document.getElementById("input-nome").value;
        if(!salaID || !nome) return;
        jogadorID = "p_" + Date.now();
        document.getElementById("display-nome").innerText = nome.toUpperCase();
        document.getElementById("screen-login").classList.remove("active");
        document.getElementById("screen-game").classList.add("active");

        const baralho = ["Político", "Político", "Miliciano", "Miliciano", "Assassino", "Assassino", "X9", "X9", "Juiz", "Juiz"];
        const c1 = baralho.splice(Math.floor(Math.random()*baralho.length), 1)[0];
        const c2 = baralho.splice(Math.floor(Math.random()*baralho.length), 1)[0];

        db.ref(`salas/${salaID}/jogadores/${jogadorID}`).set({
            nome, moedas: 1, vivo: true, cartas: [c1, c2]
        });
        ouvir();
    }

    function ouvir() {
        db.ref(`salas/${salaID}`).on("value", snap => {
            const d = snap.val();
            if(!d) return;
            render(d);
        });
    }

    function render(dados) {
        const jogadores = dados.jogadores || {};
        const eu = jogadores[jogadorID];
        if(!eu) return;

        localCards = eu.cartas || [];
        localCoins = eu.moedas || 0;
        document.getElementById("my-coins").innerText = `💰 ${localCoins}`;
        document.getElementById("my-cards").innerHTML = localCards.map(c => `
            <div class="card-large"><i class="fa-solid ${getIcon(c)}"></i><span style="font-size:10px">${c.toUpperCase()}</span></div>
        `).join('');

        // Adversários
        const container = document.getElementById("adversarios-container");
        container.innerHTML = "";
        const outros = Object.keys(jogadores).filter(id => id !== jogadorID);
        outros.forEach((id, i) => {
            const angle = (i / outros.length) * 2 * Math.PI - Math.PI/2;
            const x = Math.cos(angle) * 250;
            const y = Math.sin(angle) * 250;
            const isTurn = dados.turnoAtual === id;
            container.innerHTML += `
                <div class="player-node" style="left: calc(50% + ${x}px - 95px); top: calc(50% + ${y}px - 70px)">
                    <div class="node-box ${alvoID === id ? 'is-target' : ''} ${isTurn ? 'active-turn' : ''}" onclick="selecionarAlvo('${id}')">
                        <div class="node-name">${jogadores[id].nome}</div>
                        <div style="font-weight:900; color:var(--gold)">💰 ${jogadores[id].moedas}</div>
                        <div style="display:flex; gap:3px; justify-content:center; margin-top:5px;">
                            ${jogadores[id].cartas ? jogadores[id].cartas.map(() => `<div style="width:10px; height:15px; background:#555; border-radius:2px"></div>`).join('') : ''}
                        </div>
                    </div>
                </div>
            `;
        });

        // Logica de Turno e Timer
        const isMyTurn = dados.turnoAtual === jogadorID;
        const acaoAtiva = dados.acaoAtual && !dados.acaoAtual.finalizada;
        
        const timerUI = document.getElementById("timer-number");
        if(acaoAtiva) {
            const diff = Math.ceil((dados.acaoAtual.tempoFim - Date.now()) / 1000);
            timerUI.innerText = Math.max(0, diff);
            gestaoReacao(dados.acaoAtual, jogadores);
        } else {
            const diffTurno = Math.ceil((dados.turnoFim - Date.now()) / 1000);
            timerUI.innerText = Math.max(0, diffTurno);
            document.getElementById("action-ui").style.display = "none";
        }

        // Bloqueio de Ações (Regra das 10 moedas)
        const menu = document.getElementById("meu-menu-acoes");
        const alerta = document.getElementById("alerta-10");
        if(isMyTurn && !acaoAtiva) {
            menu.style.opacity = "1";
            menu.style.pointerEvents = "all";
            if(localCoins >= 10) {
                ["btn-t", "btn-a", "btn-p", "btn-b", "btn-x"].forEach(id => document.getElementById(id).disabled = true);
                alerta.style.display = "block";
            } else {
                ["btn-t", "btn-a", "btn-p", "btn-b", "btn-x"].forEach(id => document.getElementById(id).disabled = false);
                alerta.style.display = "none";
            }
        } else {
            menu.style.opacity = "0.3";
            menu.style.pointerEvents = "none";
        }

        // Histórico
        const logBox = document.getElementById("log-box");
        if(dados.logs) logBox.innerHTML = Object.values(dados.logs).reverse().map(l => `<div class="log-msg">${l}</div>`).join('');
    }

    function selecionarAlvo(id) { alvoID = id; }

    function fazerAcao(tipo) {
        if(["bicheiro", "investigar", "assassino"].includes(tipo) && !alvoID) return alert("Selecione um alvo!");
        
        // Desconto imediato para Vasco (Golpe de Estado)
        if(tipo === 'assassino') {
            if(localCoins < 3) return alert("Dinheiro insuficiente!");
            db.ref(`salas/${salaID}/jogadores/${jogadorID}/moedas`).transaction(m => Math.max(0, m - 3));
        }

        const acao = {
            tipo, de: jogadorID, para: alvoID,
            tempoFim: Date.now() + 15000, finalizada: false
        };

        db.ref(`salas/${salaID}/acaoAtual`).set(acao);
        db.ref(`salas/${salaID}/logs`).push(`📢 <b>${TRADUCAO[tipo]}</b> iniciado.`);
        alvoID = null;

        // Se for trampo, encerra logo o turno na lógica do servidor simulada
        if(tipo === 'trabalhar') {
            db.ref(`salas/${salaID}/acaoAtual/finalizada`).set(true);
        }
    }

    function gestaoReacao(acao, jogadores) {
        const ui = document.getElementById("action-ui");
        const btns = document.getElementById("action-btns");
        const souAlvo = acao.para === jogadorID;
        let mostrar = false;
        btns.innerHTML = "";

        if(acao.de !== jogadorID) {
            if(acao.tipo === 'politico' || acao.tipo === 'ajuda') {
                mostrar = true;
                document.getElementById("action-title").innerText = TRADUCAO[acao.tipo].toUpperCase();
                btns.innerHTML += `<button class="btn-action btn-vasco" onclick="reagir('duvidar')">DUVIDAR</button>`;
                if(acao.tipo === 'ajuda') btns.innerHTML += `<button class="btn-action" onclick="reagir('bloquear', 'Político')">POLÍTICO</button>`;
            } else if(souAlvo) {
                mostrar = true;
                document.getElementById("action-title").innerText = TRADUCAO[acao.tipo].toUpperCase();
                btns.innerHTML += `<button class="btn-action btn-vasco" onclick="reagir('duvidar')">DUVIDAR</button>`;
                if(acao.tipo === 'assassino' || acao.tipo === 'bicheiro') btns.innerHTML += `<button class="btn-action" onclick="reagir('bloquear', 'Miliciano')">MILICIANO</button>`;
                if(acao.tipo === 'investigar') btns.innerHTML += `<button class="btn-action" onclick="reagir('bloquear', 'Juiz')">JUIZ</button>`;
            }
        }

        if(mostrar) {
            ui.style.display = "block";
            btns.innerHTML += `<button class="btn-action" style="background:#333; color:white" onclick="document.getElementById('action-ui').style.display='none'">IGNORAR</button>`;
        }
    }

    function reagir(modo) {
        db.ref(`salas/${salaID}/acaoAtual`).update({ finalizada: true, contestada: true });
        db.ref(`salas/${salaID}/logs`).push(`🔥 Contestação!`);
        document.getElementById("action-ui").style.display = "none";
    }

    function resetarJogo() {
        if(confirm("Deseja mesmo reiniciar a sala?")) {
            db.ref(`salas/${salaID}`).remove();
            location.reload();
        }
    }

    // LÓGICA DE SERVIDOR SIMULADA
    setInterval(() => {
        db.ref(`salas/${salaID}`).once('value', snap => {
            const d = snap.val();
            if(!d || !d.jogadores) return;

            // Inicia Turno
            if(!d.turnoFim) {
                db.ref(`salas/${salaID}`).update({
                    turnoAtual: Object.keys(d.jogadores)[0],
                    turnoFim: Date.now() + 30000
                });
                return;
            }

            // Ação expirou
            if(d.acaoAtual && !d.acaoAtual.finalizada && Date.now() > d.acaoAtual.tempoFim) {
                concluirAcao(d.acaoAtual, d);
                return;
            }

            // Turno expirou
            if((!d.acaoAtual || d.acaoAtual.finalizada) && Date.now() > d.turnoFim) {
                passarTurno(d);
            }
        });
    }, 1500);

    function concluirAcao(acao, dados) {
        const deRef = db.ref(`salas/${salaID}/jogadores/${acao.de}`);
        if(acao.tipo === 'trabalhar') deRef.child("moedas").transaction(m => Math.min(10, (m || 0) + 1));
        if(acao.tipo === 'ajuda') deRef.child("moedas").transaction(m => Math.min(10, (m || 0) + 2));
        if(acao.tipo === 'politico') deRef.child("moedas").transaction(m => Math.min(10, (m || 0) + 3));
        
        if(acao.tipo === 'assassino') {
            // O alvo escolhe a carta no modal
            if(acao.para === jogadorID) abrirSeletorCarta("PERDER UMA CARTA", "perder");
        }

        db.ref(`salas/${salaID}/acaoAtual/finalizada`).set(true);
    }

    function abrirSeletorCarta(titulo, modo) {
        const modal = document.getElementById("card-selector-modal");
        const container = document.getElementById("selector-cards-container");
        document.getElementById("selector-title").innerText = titulo;
        container.innerHTML = "";
        
        localCards.forEach((c, i) => {
            const div = document.createElement("div");
            div.className = "card-large";
            div.innerHTML = `<i class="fa-solid ${getIcon(c)}"></i><span>${c}</span>`;
            div.onclick = () => {
                if(modo === "perder") {
                    let novas = [...localCards];
                    novas.splice(i, 1);
                    db.ref(`salas/${salaID}/jogadores/${jogadorID}/cartas`).set(novas);
                }
                modal.style.display = "none";
            };
            container.appendChild(div);
        });
        modal.style.display = "flex";
    }

    function passarTurno(dados) {
        const jids = Object.keys(dados.jogadores);
        const next = jids[(jids.indexOf(dados.turnoAtual) + 1) % jids.length];
        db.ref(`salas/${salaID}`).update({
            turnoAtual: next, turnoFim: Date.now() + 30000, acaoAtual: null
        });
    }

    function getIcon(c) {
        const icons = {"Político":"fa-user-tie", "Miliciano":"fa-shield-halved", "Assassino":"fa-skull", "X9":"fa-mask", "Juiz":"fa-gavel"};
        return icons[c] || "fa-id-card";
    }
